1    package org.firstinspires.ftc.teamcode;

7    import com.shprobotics.pestocore.drivebases.MecanumController;
8    import com.shprobotics.pestocore.drivebases.MecanumTracker;
9    import com.shprobotics.pestocore.drivebases.TeleOpController;
10   import com.shprobotics.pestocore.drivebases.ThreeWheelOdometryTracker;
11   import com.shprobotics.pestocore.geometries.Vector2D;
12   
13   @TeleOp
14   public class test extends LinearOpMode {
15   
16       @Override
17       public void runOpMode() {
18           MecanumController mecanumController = PestoFTCConfig.getMecanumController(hardwareMap);
19           ThreeWheelOdometryTracker threeWheelOdometryTracker = PestoFTCConfig.getTracker(hardwareMap);
20           //MecanumTracker mecanumTracker = PestoFTCConfig.getTracker(hardwareMap);
21           TeleOpController teleOpController = PestoFTCConfig.getTeleOpController(mecanumController, threeWheelOdometryTracker, hardwareMap);
22   
23           DcMotor viperslide = hardwareMap.get(DcMotor.class, "ViperSlide");
24           DcMotor wormgear = hardwareMap.get(DcMotor.class, "WormGear");
25           Servo claw = hardwareMap.get(Servo.class, "Claw");
26           DcMotor opodL = hardwareMap.get(DcMotor.class, "frontLeft");
27           DcMotor opodC = hardwareMap.get(DcMotor.class, "frontRight");
28           DcMotor opodR = hardwareMap.get(DcMotor.class, "backLeft");
29   
30           waitForStart();
31   
32           while (opModeIsActive()) {
33               threeWheelOdometryTracker.updateOdometry();
34               Vector2D currentPosition = threeWheelOdometryTracker.getCurrentPosition();
35               double heading = threeWheelOdometryTracker.getCurrentHeading();
36   
37               teleOpController.updateSpeed(gamepad1);
38   
39               teleOpController.driveRobotCentric(-gamepad1.left_stick_y, gamepad1.left_stick_x, gamepad1.right_stick_x);
40   //            viperslide.setMode(DcMotor.RunMode.RUN_WITHOUT_ENCODER);
41   //            viperslide.setMode(DcMotor.RunMode.STOP_AND_RESET_ENCODER);
42   
43   
44               wormgear.setPower(gamepad1.right_trigger);
45               wormgear.setPower(-gamepad1.left_trigger);
46   
47               //arm speed constraints
48               if (gamepad1.right_stick_y > 0.5) {
49                   viperslide.setPower(0.5);
50                   }
51               //if gamepad1.right_stick_y is between 10% and 50%
52               else if (gamepad1.dpad_up) {
53                   viperslide.setPower(0.5);
54               }
55               else if (gamepad1.dpad_down){
56                   viperslide.setPower(-0.5);
57               }
58   
59   
60               if (gamepad1.x) {
61                       claw.setPosition(0.5);
62                   }
63               if (gamepad1.y){
64                   claw.setPosition(1.0);
65               }
66   
67               telemetry.addData("x", currentPosition.getX());
68               telemetry.addData("y", currentPosition.getY());
69               telemetry.addData("rotation", heading);
70               telemetry.addData("OPOD L", opodL.getCurrentPosition());
71               telemetry.addData("OPOD C", opodC.getCurrentPosition());
72               telemetry.addData("OPOD R", opodR.getCurrentPosition());
73               telemetry.addData("Viper Slide: ", viperslide.getCurrentPosition());
74               telemetry.update();
75   
76   
77               if (gamepad1.b) {
78                   threeWheelOdometryTracker.reset();
79                   teleOpController.resetIMU();
80               }
81           }
82       }
83   }
